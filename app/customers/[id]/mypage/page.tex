"use client"

import { useEffect, useMemo, useState } from "react"
import { useParams, useRouter } from "next/navigation"
import { supabase } from "@/lib/supabase"

type Customer = {
  id: string
  name: string | null
  phone: string | null
  line: string | null
  memo: string | null
  created_at: string | null
}

type Visit = {
  id: string
  menu: string | null
  color: string | null
  memo: string | null
  price: number | null
  created_at: string | null
  photo_urls: string[] | null
}

type Reservation = {
  id: string
  customer_id: string | null
  menu: string | null
  memo: string | null
  reservation_date: string | null
  created_at: string | null
}

const GOOGLE_REVIEW_URL = "https://g.page/r/REPLACE_WITH_YOUR_REVIEW_LINK/review"

export default function CustomerMyPage() {
  const router = useRouter()
  const params = useParams()

  const customerId = useMemo(() => {
    const raw = params?.id
    return Array.isArray(raw) ? raw[0] : raw ?? ""
  }, [params])

  const [customer, setCustomer] = useState<Customer | null>(null)
  const [visits, setVisits] = useState<Visit[]>([])
  const [reservations, setReservations] = useState<Reservation[]>([])
  const [loading, setLoading] = useState(true)
  const [errorMessage, setErrorMessage] = useState("")
  const [selectedRating, setSelectedRating] = useState<number | null>(null)

  useEffect(() => {
    if (!customerId) return
    void loadData()
  }, [customerId])

  async function loadData() {
    setLoading(true)
    setErrorMessage("")

    const { data: customerData, error: customerError } = await supabase
      .from("customers")
      .select("id, name, phone, line, memo, created_at")
      .eq("id", customerId)
      .single()

    if (customerError) {
      setErrorMessage("顧客情報の取得に失敗しました。")
      setLoading(false)
      return
    }

    const { data: visitsData } = await supabase
      .from("visits")
      .select("id, menu, color, memo, price, created_at, photo_urls")
      .eq("customer_id", customerId)
      .order("created_at", { ascending: false })

    const { data: reservationsData } = await supabase
      .from("reservations")
      .select("id, customer_id, menu, memo, reservation_date, created_at")
      .eq("customer_id", customerId)
      .order("reservation_date", { ascending: true })

    setCustomer(customerData)
    setVisits(visitsData ?? [])
    setReservations(reservationsData ?? [])
    setLoading(false)
  }

  function formatDate(date: string | null) {
    if (!date) return "未登録"
    const d = new Date(date)
    if (Number.isNaN(d.getTime())) return "未登録"
    return `${d.getFullYear()}/${d.getMonth() + 1}/${d.getDate()}`
  }

  function formatDateTime(date: string | null) {
    if (!date) return "未登録"
    const d = new Date(date)
    if (Number.isNaN(d.getTime())) return "未登録"

    const y = d.getFullYear()
    const m = d.getMonth() + 1
    const day = d.getDate()
    const hh = String(d.getHours()).padStart(2, "0")
    const mm = String(d.getMinutes()).padStart(2, "0")

    return `${y}/${m}/${day} ${hh}:${mm}`
  }

  function getDaysFromToday(date: string | null) {
    if (!date) return null
    const base = new Date(date)
    if (Number.isNaN(base.getTime())) return null

    const today = new Date()
    today.setHours(0, 0, 0, 0)
    base.setHours(0, 0, 0, 0)

    const diff = today.getTime() - base.getTime()
    return Math.floor(diff / (1000 * 60 * 60 * 24))
  }

  function getDaysUntil(date: string | null) {
    if (!date) return null
    const target = new Date(date)
    if (Number.isNaN(target.getTime())) return null

    const today = new Date()
    today.setHours(0, 0, 0, 0)
    target.setHours(0, 0, 0, 0)

    const diff = target.getTime() - today.getTime()
    return Math.floor(diff / (1000 * 60 * 60 * 24))
  }

  const now = Date.now()

  const lastVisit = visits.length > 0 ? visits[0] : null

  const upcomingReservations = reservations.filter((reservation) => {
    if (!reservation.reservation_date) return false
    const time = new Date(reservation.reservation_date).getTime()
    if (Number.isNaN(time)) return false
    return time >= now
  })

  const nextReservation =
    upcomingReservations.length > 0 ? upcomingReservations[0] : null

  const daysSinceLastVisit = getDaysFromToday(lastVisit?.created_at ?? null)
  const daysUntilNextReservation = getDaysUntil(
    nextReservation?.reservation_date ?? null
  )

  function handleRatingClick(rating: number) {
    setSelectedRating(rating)
  }

  function handleGoogleReview() {
    window.open(GOOGLE_REVIEW_URL, "_blank", "noopener,noreferrer")
  }

  if (loading) {
    return <div className="p-6">読み込み中...</div>
  }

  if (!customerId) {
    return <div className="p-6">顧客IDが見つかりません</div>
  }

  if (!customer) {
    return <div className="p-6">顧客が見つかりません</div>
  }

  return (
    <div className="min-h-screen bg-gray-50 p-4 md:p-6">
      <div className="mx-auto max-w-3xl space-y-6">
        <div className="rounded-3xl bg-white p-6 shadow-sm border">
          <div className="space-y-2 text-center">
            <div className="text-sm text-gray-500">Aily Nail Studio</div>
            <h1 className="text-2xl font-bold">
              {customer.name ?? "お客様"}さま
            </h1>
            <p className="text-sm text-gray-600">
              ご来店ありがとうございます。ご予約確認や口コミ投稿にご利用ください。
            </p>
          </div>
        </div>

        {errorMessage ? (
          <div className="rounded-2xl border border-red-200 bg-red-50 px-4 py-3 text-sm text-red-700">
            {errorMessage}
          </div>
        ) : null}

        <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
          <div className="rounded-3xl border bg-white p-5 shadow-sm">
            <div className="text-sm text-gray-500">前回のご来店</div>
            <div className="mt-2 text-3xl font-bold">
              {daysSinceLastVisit === null ? "-" : `${daysSinceLastVisit}日`}
            </div>
            <div className="mt-2 text-sm text-gray-600">
              来店日：{formatDate(lastVisit?.created_at ?? null)}
            </div>
            <div className="mt-2 text-sm text-gray-600">
              メニュー：{lastVisit?.menu ?? "未登録"}
            </div>
          </div>

          <div className="rounded-3xl border bg-white p-5 shadow-sm">
            <div className="text-sm text-gray-500">次回のご予約</div>
            <div className="mt-2 text-lg font-bold">
              {nextReservation
                ? formatDateTime(nextReservation.reservation_date)
                : "予約はありません"}
            </div>
            <div className="mt-2 text-sm text-gray-600">
              {nextReservation
                ? daysUntilNextReservation === 0
                  ? "本日のご予約です"
                  : daysUntilNextReservation === null
                  ? "日数未計算"
                  : `${daysUntilNextReservation}日後のご予約です`
                : "次回のご予約をお待ちしております"}
            </div>
            <div className="mt-2 text-sm text-gray-600">
              メニュー：{nextReservation?.menu ?? "未登録"}
            </div>
          </div>
        </div>

        <div className="rounded-3xl border bg-white p-6 shadow-sm space-y-5">
          <div className="text-center">
            <h2 className="text-xl font-bold">今回のご満足度はいかがでしたか？</h2>
            <p className="mt-1 text-sm text-gray-600">
              ボタンを押してお聞かせください
            </p>
          </div>

          <div className="grid grid-cols-5 gap-2">
            {[1, 2, 3, 4, 5].map((rating) => (
              <button
                key={rating}
                onClick={() => handleRatingClick(rating)}
                className={`rounded-2xl border px-2 py-4 text-center transition ${
                  selectedRating === rating
                    ? "border-pink-500 bg-pink-50"
                    : "border-gray-200 bg-white"
                }`}
              >
                <div className="text-2xl">★</div>
                <div className="mt-1 text-sm font-medium">{rating}</div>
              </button>
            ))}
          </div>

          {!selectedRating ? (
            <div className="rounded-2xl bg-gray-50 p-4 text-sm text-gray-600">
              星を選ぶと次の案内が表示されます。
            </div>
          ) : null}

          {selectedRating >= 4 ? (
            <div className="rounded-2xl border border-green-200 bg-green-50 p-5 space-y-4">
              <div>
                <div className="text-lg font-bold text-green-800">
                  ありがとうございます！
                </div>
                <div className="mt-1 text-sm text-green-700">
                  よろしければ Google の口コミ投稿にご協力ください。
                </div>
              </div>

              <button
                onClick={handleGoogleReview}
                className="w-full rounded-2xl bg-green-600 px-4 py-3 text-white font-medium"
              >
                Google口コミを投稿する
              </button>

              <div className="text-xs text-green-700">
                ※ 店舗の実際の口コミURLを設定するとそのまま使えます
              </div>
            </div>
          ) : null}

          {selectedRating <= 3 ? (
            <div className="rounded-2xl border border-amber-200 bg-amber-50 p-5 space-y-3">
              <div className="text-lg font-bold text-amber-800">
                ご意見ありがとうございます
              </div>
              <div className="text-sm text-amber-700">
                より良いサービスのため、スタッフへご要望をお聞かせください。
              </div>
              <div className="rounded-2xl bg-white p-4 text-sm text-gray-700">
                店頭スタッフ、または LINE にてお気軽にお知らせください。
              </div>
            </div>
          ) : null}
        </div>

        <div className="rounded-3xl border bg-white p-6 shadow-sm space-y-4">
          <h2 className="text-lg font-bold">ご案内</h2>

          <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
            <button
              onClick={() =>
                router.push(`/reservations/new?customer_id=${customerId}`)
              }
              className="rounded-2xl bg-blue-600 px-4 py-3 text-white font-medium"
            >
              次回予約を登録
            </button>

            <button
              onClick={() => router.push(`/customers/${customerId}`)}
              className="rounded-2xl border px-4 py-3 font-medium"
            >
              顧客詳細へ戻る
            </button>
          </div>
        </div>

        <div className="rounded-3xl border bg-white p-6 shadow-sm space-y-4">
          <h2 className="text-lg font-bold">最近の来店履歴</h2>

          {visits.length === 0 ? (
            <div className="rounded-2xl bg-gray-50 p-4 text-sm text-gray-500">
              来店履歴はまだありません
            </div>
          ) : (
            <div className="space-y-3">
              {visits.slice(0, 3).map((visit) => (
                <div key={visit.id} className="rounded-2xl border p-4">
                  <div className="flex items-start justify-between gap-3">
                    <div>
                      <div className="text-sm text-gray-500">来店日</div>
                      <div className="font-semibold">
                        {formatDate(visit.created_at)}
                      </div>
                    </div>
                    <div className="text-right">
                      <div className="text-sm text-gray-500">料金</div>
                      <div className="font-semibold">
                        ¥{(visit.price ?? 0).toLocaleString()}
                      </div>
                    </div>
                  </div>

                  <div className="mt-3 grid grid-cols-1 md:grid-cols-2 gap-3 text-sm">
                    <div className="rounded-xl bg-gray-50 p-3">
                      <div className="text-gray-500">メニュー</div>
                      <div className="mt-1 font-medium">{visit.menu ?? "-"}</div>
                    </div>
                    <div className="rounded-xl bg-gray-50 p-3">
                      <div className="text-gray-500">カラー</div>
                      <div className="mt-1 font-medium">{visit.color ?? "-"}</div>
                    </div>
                  </div>

                  <div className="mt-3 rounded-xl bg-gray-50 p-3 text-sm">
                    <div className="text-gray-500">メモ</div>
                    <div className="mt-1 whitespace-pre-wrap">
                      {visit.memo ?? "-"}
                    </div>
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      </div>
    </div>
  )
}
